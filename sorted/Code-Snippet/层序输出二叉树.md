# Binary Tree Layer Print

#二叉树  #cpp #层序遍历  #用于leetcode本地调试 

```c++
void layer_print_tree(TreeNode *root) {  
  std::queue<TreeNode *> tq;  
  tq.push(root);  
  int j = 0;  
  while (!tq.empty()) {  
    TreeNode *node = tq.front();  
    tq.pop();  
    if (node) {  
      std::cout << node->val << " ";  
      tq.push(node->left);  
      tq.push(node->right);  
    } else {  
      std::cout << "NULL" << " ";  
    }  
  }  
}
```
